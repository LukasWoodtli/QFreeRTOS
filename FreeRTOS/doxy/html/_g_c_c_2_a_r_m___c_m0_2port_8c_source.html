134 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
135 {
136  /* Simulate the stack frame as it would be created by a context switch
137  interrupt. */
138  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
139  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
140  pxTopOfStack--;
141  *pxTopOfStack = ( StackType_t ) pxCode; /* PC */
142  pxTopOfStack--;
143  *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* LR */
144  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
145  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
146  pxTopOfStack -= 8; /* R11..R4. */
147 
148  return pxTopOfStack;
149 }

152 static void prvTaskExitError( void )
153 {
154  /* A function that implements a task must not exit or attempt to return to
155  its caller as there is nothing to return to. If a task wants to exit it
156  should instead call vTaskDelete( NULL ).

158  Artificially force an assert() to be triggered if configASSERT() is
159  defined, then stop here so application writers can catch the error. */
160  configASSERT( uxCriticalNesting == ~0UL );
161  portDISABLE_INTERRUPTS();
162  for( ;; );
163 }

173 void vPortStartFirstTask( void )
174 {
175  /* The MSP stack is not reset as, unlike on M3/4 parts, there is no vector
176  table offset register that can be used to locate the initial stack value.
177  Not all M0 parts have the application vector table at address 0. */
178  __asm volatile(
179  " ldr r2, pxCurrentTCBConst2 \n" /* Obtain location of pxCurrentTCB. */
180  " ldr r3, [r2] \n"
181  " ldr r0, [r3] \n" /* The first item in pxCurrentTCB is the task top of stack. */
182  " add r0, #32 \n" /* Discard everything up to r0. */
183  " msr psp, r0 \n" /* This is now the new top of stack to use in the task. */
184  " movs r0, #2 \n" /* Switch to the psp stack. */
185  " msr CONTROL, r0 \n"
186  " pop {r0-r5} \n" /* Pop the registers that are saved automatically. */
187  " mov lr, r5 \n" /* lr is now in r5. */
188  " cpsie i \n" /* The first task has its context and interrupts can be enabled. */
189  " pop {pc} \n" /* Finally, pop the PC to jump to the user defined task code. */
190  " \n"
191  " .align 2 \n"
192  "pxCurrentTCBConst2: .word pxCurrentTCB "
193  );
194 }

200 BaseType_t xPortStartScheduler( void )
201 {
202  /* Make PendSV, CallSV and SysTick the same priroity as the kernel. */
203  *(portNVIC_SYSPRI2) |= portNVIC_PENDSV_PRI;
204  *(portNVIC_SYSPRI2) |= portNVIC_SYSTICK_PRI;
205 
206  /* Start the timer that generates the tick ISR. Interrupts are disabled
207  here already. */
208  prvSetupTimerInterrupt();
209 
210  /* Initialise the critical nesting count ready for the first task. */
211  uxCriticalNesting = 0;
212 
213  /* Start the first task. */
214  vPortStartFirstTask();
215 
216  /* Should never get here as the tasks will now be executing! Call the task
217  exit error function to prevent compiler warnings about a static function
218  not being called in the case that the application writer overrides this
219  functionality by defining configTASK_RETURN_ADDRESS. */
220  prvTaskExitError();
221 
222  /* Should not get here! */
223  return 0;
224 }

292 void xPortPendSVHandler( void )
293 {
294  /* This is a naked function. */
295 
296  __asm volatile
297  (
298  " mrs r0, psp \n"
299  " \n"
300  " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
301  " ldr r2, [r3] \n"
302  " \n"
303  " sub r0, r0, #32 \n" /* Make space for the remaining low registers. */
304  " str r0, [r2] \n" /* Save the new top of stack. */
305  " stmia r0!, {r4-r7} \n" /* Store the low registers that are not saved automatically. */
306  " mov r4, r8 \n" /* Store the high registers. */
307  " mov r5, r9 \n"
308  " mov r6, r10 \n"
309  " mov r7, r11 \n"
310  " stmia r0!, {r4-r7} \n"
311  " \n"
312  " push {r3, r14} \n"
313  " cpsid i \n"
314  " bl vTaskSwitchContext \n"
315  " cpsie i \n"
316  " pop {r2, r3} \n" /* lr goes in r3. r2 now holds tcb pointer. */
317  " \n"
318  " ldr r1, [r2] \n"
319  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
320  " add r0, r0, #16 \n" /* Move to the high registers. */
321  " ldmia r0!, {r4-r7} \n" /* Pop the high registers. */
322  " mov r8, r4 \n"
323  " mov r9, r5 \n"
324  " mov r10, r6 \n"
325  " mov r11, r7 \n"
326  " \n"
327  " msr psp, r0 \n" /* Remember the new top of stack for the task. */
328  " \n"
329  " sub r0, r0, #32 \n" /* Go back for the low registers that are not automatically restored. */
330  " ldmia r0!, {r4-r7} \n" /* Pop low registers. */
331  " \n"
332  " bx r3 \n"
333  " \n"
334  " .align 2 \n"
335  "pxCurrentTCBConst: .word pxCurrentTCB "
336  );
337 }

340 void xPortSysTickHandler( void )
341 {
342 uint32_t ulPreviousMask;
343 
344  ulPreviousMask = portSET_INTERRUPT_MASK_FROM_ISR();
345  {
346  /* Increment the RTOS tick. */
347  if( xTaskIncrementTick() != pdFALSE )
348  {
349  /* Pend a context switch. */
350  *(portNVIC_INT_CTRL) = portNVIC_PENDSVSET;
351  }
352  }
353  portCLEAR_INTERRUPT_MASK_FROM_ISR( ulPreviousMask );
354 }
