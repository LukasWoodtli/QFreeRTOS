150 /* Don't yield if inside a critical section - instead hold the yield pending
151 so it is performed when the critical section is exited. */
152 #define portYIELD() \
153 { \
154 extern volatile uint32_t ulCriticalNesting; \
155 extern volatile uint32_t ulPortYieldPending; \
156  if( ulCriticalNesting != 0 ) \
157  { \
158  ulPortYieldPending = pdTRUE; \
159  } \
160  else \
161  { \
162  __asm volatile( portYIELD_INTERRUPT ); \
163  } \
164 }

253 #ifndef configUSE_PORT_OPTIMISED_TASK_SELECTION
254  /* The FreeRTOS scheduling algorithm selects the task that will enter the
255  Running state. configUSE_PORT_OPTIMISED_TASK_SELECTION is used to set how
256  that is done.
257 
258  If configUSE_PORT_OPTIMISED_TASK_SELECTION is set to 0 then the task to
259  enter the Running state is selected using a portable algorithm written in
260  C. This is the slowest method, but the algorithm does not restrict the
261  maximum number of unique RTOS task priorities that are available.
262 
263  If configUSE_PORT_OPTIMISED_TASK_SELECTION is set to 1 then the task to
264  enter the Running state is selected using a single assembly instruction.
265  This is the fastest method, but restricts the maximum number of unique RTOS
266  task priorities to 32 (the same task priority can be assigned to any number
267  of RTOS tasks). */
268  #warning configUSE_PORT_OPTIMISED_TASK_SELECTION was not defined in FreeRTOSConfig.h and has been defaulted to 1
269  #define configUSE_PORT_OPTIMISED_TASK_SELECTION 1
270 #endif

272 #ifndef configUSE_COMMON_INTERRUPT_ENTRY_POINT
273  /* There are two ways of implementing interrupt handlers:
274 
275  1) As standard C functions -
276 
277  This method can only be used if configUSE_COMMON_INTERRUPT_ENTRY_POINT
278  is set to 1. The C function is installed using
279  xPortRegisterCInterruptHandler().
280 
281  This is the simplest of the two methods but incurs a slightly longer

285  portFREERTOS_INTERRUPT_ENTRY and portFREERTOS_INTERRUPT_EXIT macros.
286 
287  This method can always be used. It is slightly more complex than
