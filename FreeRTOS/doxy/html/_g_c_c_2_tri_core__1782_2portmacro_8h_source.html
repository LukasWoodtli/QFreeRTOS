155 #define portDISABLE_INTERRUPTS() { \
156  uint32_t ulICR; \
157  _disable(); \
158  ulICR = _mfcr( $ICR ); /* Get current ICR value. */ \
159  ulICR &= ~portCCPN_MASK; /* Clear down mask bits. */ \
160  ulICR |= configMAX_SYSCALL_INTERRUPT_PRIORITY; /* Set mask bits to required priority mask. */ \
161  _mtcr( $ICR, ulICR ); /* Write back updated ICR. */ \
162  _isync(); \
163  _enable(); \
164  }
