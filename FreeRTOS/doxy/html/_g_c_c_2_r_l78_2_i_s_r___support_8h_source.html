70 /*
71  * portSAVE_CONTEXT MACRO
72  * Saves the context of the general purpose registers, CS and ES (only in far
73  * memory mode) registers the usCriticalNesting Value and the Stack Pointer
74  * of the active Task onto the task stack
75  */
76  .macro portSAVE_CONTEXT
77 
78  SEL RB0
79 
80  /* Save AX Register to stack. */
81  PUSH AX
82  PUSH HL
83  /* Save CS register. */
84  MOV A, CS
85  XCH A, X
86  /* Save ES register. */
87  MOV A, ES
88  PUSH AX
89  /* Save the remaining general purpose registers from bank 0. */
90  PUSH DE
91  PUSH BC
92  /* Save the other register banks - only necessary in the GCC port. */
93  SEL RB1
94  PUSH AX
95  PUSH BC
96  PUSH DE
97  PUSH HL
98  SEL RB2
99  PUSH AX
100  PUSH BC
101  PUSH DE
102  PUSH HL
103  /* Registers in bank 3 are for ISR use only so don't need saving. */
104  SEL RB0
105  /* Save the usCriticalNesting value. */
106  MOVW AX, !_usCriticalNesting
107  PUSH AX
108  /* Save the Stack pointer. */
109  MOVW AX, !_pxCurrentTCB
110  MOVW HL, AX
111  MOVW AX, SP
112  MOVW [HL], AX
113  /* Switch stack pointers. */
114  movw sp,#_stack /* Set stack pointer */
115 
116  .endm

119 /*
120  * portRESTORE_CONTEXT MACRO
121  * Restores the task Stack Pointer then use this to restore usCriticalNesting,
122  * general purpose registers and the CS and ES (only in far memory mode)
123  * of the selected task from the task stack
124  */
125 .macro portRESTORE_CONTEXT MACRO
126  SEL RB0
127  /* Restore the Stack pointer. */
128  MOVW AX, !_pxCurrentTCB
129  MOVW HL, AX
130  MOVW AX, [HL]
131  MOVW SP, AX
132  /* Restore usCriticalNesting value. */
133  POP AX
134  MOVW !_usCriticalNesting, AX
135  /* Restore the alternative register banks - only necessary in the GCC
136  port. Register bank 3 is dedicated for interrupts use so is not saved or
137  restored. */
138  SEL RB2
139  POP HL
140  POP DE
141  POP BC
142  POP AX
143  SEL RB1
144  POP HL
145  POP DE
146  POP BC
147  POP AX
148  SEL RB0
149  /* Restore the necessary general purpose registers. */
150  POP BC
151  POP DE
152  /* Restore the ES register. */
153  POP AX
154  MOV ES, A
155  /* Restore the CS register. */
156  XCH A, X
157  MOV CS, A
158  /* Restore general purpose register HL. */
159  POP HL
160  /* Restore AX. */
161  POP AX
162 
163  .endm
