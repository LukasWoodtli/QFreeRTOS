119 /*
120  * portRESTORE_CONTEXT, portRESTORE_CONTEXT, portENTER_SWITCHING_ISR
121  * and portEXIT_SWITCHING_ISR can only be called from ARM mode, but
122  * are included here for efficiency. An attempt to call one from
123  * THUMB mode code will result in a compile time error.
124  */
125 
126 #define portRESTORE_CONTEXT() \
127 { \
128 extern volatile void * volatile pxCurrentTCB; \
129 extern volatile uint32_t ulCriticalNesting; \
130  \
131  /* Set the LR to the task stack. */ \
132  __asm volatile ( \
133  "LDR R0, =pxCurrentTCB \n\t" \
134  "LDR R0, [R0] \n\t" \
135  "LDR LR, [R0] \n\t" \
136  \
137  /* The critical nesting depth is the first item on the stack. */ \
138  /* Load it into the ulCriticalNesting variable. */ \
139  "LDR R0, =ulCriticalNesting \n\t" \
140  "LDMFD LR!, {R1} \n\t" \
141  "STR R1, [R0] \n\t" \
142  \
143  /* Get the SPSR from the stack. */ \
144  "LDMFD LR!, {R0} \n\t" \
145  "MSR SPSR, R0 \n\t" \
146  \
147  /* Restore all system mode registers for the task. */ \
148  "LDMFD LR, {R0-R14}^ \n\t" \
149  "NOP \n\t" \
150  \
151  /* Restore the return address. */ \
152  "LDR LR, [LR, #+60] \n\t" \
153  \
154  /* And return - correcting the offset in the LR to obtain the */ \
155  /* correct address. */ \
156  "SUBS PC, LR, #4 \n\t" \
157  ); \
158  ( void ) ulCriticalNesting; \
159  ( void ) pxCurrentTCB; \
160 }

163 #define portSAVE_CONTEXT() \
164 { \
165 extern volatile void * volatile pxCurrentTCB; \
166 extern volatile uint32_t ulCriticalNesting; \
167  \
168  /* Push R0 as we are going to use the register. */ \
169  __asm volatile ( \
170  "STMDB SP!, {R0} \n\t" \
171  \
172  /* Set R0 to point to the task stack pointer. */ \
173  "STMDB SP,{SP}^ \n\t" \
174  "NOP \n\t" \
175  "SUB SP, SP, #4 \n\t" \
176  "LDMIA SP!,{R0} \n\t" \
177  \
178  /* Push the return address onto the stack. */ \
179  "STMDB R0!, {LR} \n\t" \
180  \
181  /* Now we have saved LR we can use it instead of R0. */ \
182  "MOV LR, R0 \n\t" \
183  \
184  /* Pop R0 so we can save it onto the system mode stack. */ \
185  "LDMIA SP!, {R0} \n\t" \
186  \
187  /* Push all the system mode registers onto the task stack. */ \
188  "STMDB LR,{R0-LR}^ \n\t" \
189  "NOP \n\t" \
190  "SUB LR, LR, #60 \n\t" \
191  \
192  /* Push the SPSR onto the task stack. */ \
193  "MRS R0, SPSR \n\t" \
194  "STMDB LR!, {R0} \n\t" \
195  \
196  "LDR R0, =ulCriticalNesting \n\t" \
197  "LDR R0, [R0] \n\t" \
198  "STMDB LR!, {R0} \n\t" \
199  \
200  /* Store the new top of stack for the task. */ \
201  "LDR R0, =pxCurrentTCB \n\t" \
202  "LDR R0, [R0] \n\t" \
203  "STR LR, [R0] \n\t" \
204  ); \
205  ( void ) ulCriticalNesting; \
206  ( void ) pxCurrentTCB; \
207 }
