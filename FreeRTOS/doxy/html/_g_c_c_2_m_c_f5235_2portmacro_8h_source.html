92 /*
93  * This function must be called when the current state of the active task
94  * should be stored. It must be called immediately after exception
95  * processing from the CPU, i.e. there exists a Coldfire exception frame at
96  * the current position in the stack. The function reserves space on
97  * the stack for the CPU registers and other task dependent values (e.g
98  * ulCriticalNesting) and updates the top of the stack in the TCB.
99  */
100 #define portSAVE_CONTEXT() \
101  asm volatile ( /* reserve space for task state. */ \
102  "lea.l (-64, %sp), %sp\n\t" \
103  /* push data register %d0-%d7/%a0-%a6 on stack. */ \
104  "movem.l %d0-%d7/%a0-%a6, (%sp)\n\t" \
105  /* push ulCriticalNesting counter on stack. */ \
106  "lea.l (60, %sp), %a0\n\t" \
107  "move.l ulCriticalNesting, (%a0)\n\t" \
108  /* set the new top of the stack in the TCB. */ \
109  "move.l pxCurrentTCB, %a0\n\t" \
110  "move.l %sp, (%a0)");
111 
112 /*.
113  * This function restores the current active and continues its execution.
114  * It loads the current TCB and restores the processor registers, the
115  * task dependent values (e.g ulCriticalNesting). Finally execution
116  * is continued by executing an rte instruction.
117  */
118 #define portRESTORE_CONTEXT() \
119  asm volatile ( "move.l pxCurrentTCB, %sp\n\t" \
120  "move.l (%sp), %sp\n\t" \
121  /* stack pointer now points to the saved registers. */ \
122  "movem.l (%sp), %d0-%d7/%a0-%a6\n\t" \
123  /* restore ulCriticalNesting counter from stack. */ \
124  "lea.l (%sp, 60), %sp\n\t" \
125  "move.l (%sp)+, ulCriticalNesting\n\t" \
126  /* stack pointer now points to exception frame. */ \
127  "rte\n\t" );
